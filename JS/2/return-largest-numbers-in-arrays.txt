


function largestOfFour(arr) {

        let newArr = []

        for (let i = 0; i < arr.length; i++) {

            let maxNum = arr[i][0]
            for (let b = 0; b < arr[i].length; b++) {
                if (arr[i][b] > maxNum) {
                    maxNum = arr[i][b]
                }
            }
            newArr.push(maxNum)
        }
        console.log(newArr)
        return newArr
}

largestOfFour([[4, 5, 1, 3], [13, 27, 18, 26], [32, 35, 37, 39], [1000, 1001, 857, 1]]);
